public void main (String[] args) {
    robot.moveForward();
    robot.moveForward();
    if (robot.seesRock()) {
        robot.moveForward();
        robot.turnLeft();
    }
    else {
        robot.turnLeft();
        robot.moveForward();
    }
    robot.moveForward();
    robot.turnLeft();
    robot.moveForward();
    if (robot.seesRock()) {
        robot.turnRight();
        robot.moveForward();
    }
    robot.moveForward();
    robot.moveForward();
}
