public void main (String[] args) {
    int ores = 0;
    int oreGoal = 2;
    while (ores < oreGoal) {
        if (robot.seesRock()) {
            robot.turnRight();
            robot.moveForward();
        }
        else {
            robot.turnLeft();
            if (robot.seesRock()) {
                robot.turnLeft();
            }
            else {
                robot.moveFoward();
            }
        }
        robot.turnRight();
        robot.moveForward();
        if (robot.pickedOre()) {
            ores ++;
        }
    }
